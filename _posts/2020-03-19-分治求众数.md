**用分治策略实现寻找众数及其重数。
首先给待寻数组排序。找出中位数并且确定中位数的个数和其左右边界。
如果左边界左边数字的个数比中位数的个数大，说明众数可能在左边，重复之前的的步骤。如果右边界右边数字的个数比中位数的个数大，说明众数可能在右边，重复之前的的步骤。（递归解决）
最终可寻找得到众数和其重数。**

``` javascript
#include<iostream>
#include<cstdio>
#include<algorithm>
#include<map>
using namespace std;

map<int, int> m;

// 确定中位数的个数和其左右边界
void Div(int M[], int n, int& Left, int& Right)
{//求与M[Mid]相同的数的个数
    int Mid = n / 2;
    for (Left = 0; Left < n; ++Left)
    {//左
        if (M[Left] == M[Mid])
            break;
    }
    for (Right = Left + 1; Right < n; ++Right)
    {//右
        if (M[Right] != M[Mid])
            break;
    }
}

void Get(int& Mid, int& MaxCnt, int M[], int n)
{// Num 众数。 MaxCnt 重数
    int Num = n / 2;
    int Left, Right;
    Div(M, n, Left, Right);//进行分割
    int Cnt = Right - Left;
    if (Cnt > MaxCnt)
    {// 更新众数
        MaxCnt = Cnt;
        Mid = M[Num];
        m.clear();
        m[Mid] = MaxCnt;
    }
    else if (Cnt == MaxCnt)
    {
        Mid = M[Num];
        m[Mid] = MaxCnt;
    }
    if (Left >= MaxCnt)
    {//如果左边界左边数字的个数比中位数的个数大，说明众数可能在左边
        Get(Mid, MaxCnt, M, Left);
    }   
    if (n - Right >= MaxCnt)
    {// 如果右边界右边数字的个数比中位数的个数大，说明众数可能在右边
        Get(Mid, MaxCnt, M + Right, n - Right);
    }
}

int main(void)
{
    int n;
    cin >> n;
    int* M = new int[n * sizeof(int)];
    for (int i = 0; i < n; i++)
        cin >> M[i];
    cout << endl;
    //初始化
    sort(M, M + n);//排序

    int Max = 0;
    int N = 0;
    Get(N, Max, M, n);
    map<int, int>::iterator Num;
    int Count = 0;
    for (Num = m.begin(); Num != m.end(); Num++)
    {
        Count += Num->second;
    }
    if (Count != n)
    {
        for (Num = m.begin(); Num != m.end(); Num++)
        {
            cout << Num->first << endl << Num->second << endl;
        }
    }
    else
    {
        cout << "NULL!" << endl;
    }
    delete[] M;

    return 0;
}
```
